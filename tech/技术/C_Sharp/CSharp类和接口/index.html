    CSharp类和接口

    C#中的构造函数与析构函数

    using System;
namespace LineApplication
{
   class Line
   {
      private double length;   //线条的长度
      public Line()  // 构造函数
      {
         Console.WriteLine("对象已创建");
      }
      ~Line() //析构函数
      {
         Console.WriteLine("对象已删除");
      }

      public void setLength( double len )
      {
         length = len;
      }
      public double getLength()
      {
         return length;
      }

      static void Main(string[] args)
      {
         Line line = new Line();
         // 设置线条长度
         line.setLength(6.0);
         Console.WriteLine("线条的长度： {0}", line.getLength());           
      }
   }
}

    C#接口代码

    • 接口定义了所有类继承接口时应遵循的语法合同。接口定义了语法合同 "是什么" 部分，派生类定义了语法合同 "怎么做" 部分。

    • 接口定义了属性、方法和事件，这些都是接口的成员。接口只包含了成员的声明。成员的定义是派生类的责任。接口提供了派生类应遵循的标准结构。

    • 接口使得实现接口的类或结构在形式上保持一致。

    using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
namespace ConsoleApplication_interface
{
    //公共接口: "动物"
    public interface IAnimal
    {  void Behavior();  }

    //类: 狗
    public class Dog : IAnimal
    {        
        public void Behavior()
        { Console.WriteLine("我晚上睡觉,白天活动"); }
    }

    //类: 猫
    public class Cat : IAnimal
    {        
        public void Behavior()
        { Console.WriteLine("我白天睡觉,晚上活动"); }
    }

    //类: 龟
    public class Tortoise : IAnimal
    {
        public void Behavior()
        { Console.WriteLine("我可以不活动，一睡就睡五千年!"); }
    }


    class Program
    {
        static void Main(string[] args)
        {
            Dog myDog = new Dog();
            myDog.Behavior(); //输出: "我晚上睡觉,白天活动"

            Cat myCat = new Cat();
            myCat.Behavior(); //输出: "我白天睡觉,晚上活动"

            Tortoise tortoise = new Tortoise();
            tortoise.Behavior(); 
            Console.Read();
        }
    }
}
