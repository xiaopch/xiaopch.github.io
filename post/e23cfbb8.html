新建自己的package

//$GOPATH/src/mymath/sqrt.go源码如下：

package mymath

func Sqrt(x float64) float64 {
    z := 0.0
    for i := 0; i < 1000; i++ {
        z -= (z*z - x) / (2 * x)
    }
    return z
}

使用自己的package

// $GOPATH/src/mathapp/main.go源码：

package main

import (
    mymath
    fmt
)

func main() {
    fmt.Printf(Hello, world.  Sqrt(2) = %v\n, mymath.Sqrt(2))
}
